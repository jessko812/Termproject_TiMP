20bool NetworkManager::connect() {

79bool NetworkManager::authenticate(const std::string& login, const std::string& password) {
80 try {
81 //Отправка логина
82 sendString(login);
83 std::cout << "Отправлен логин: " << login << std::endl;
84
85 //Получение соли от сервера или ERR
86 std::string salt_response = receiveString();
87 std::cout << "Получена соль от сервера " <<std::endl;
88
89 if (salt_response == "ERR") {
90 std::cerr << "Сервер вернул ошибку идентификации" << std::endl;
91 return false;
92 }
93
94 //Вычисление хеша SHA-224(SALT16 || PASSWORD)
95 std::string hash = LinuxAuthenticator::computeSHA224(salt_response, password);
96
97 // ПРЕОБРАЗОВАНИЕ В ВЕРХНИЙ РЕГИСТР
98 std::transform(hash.begin(), hash.end(), hash.begin(),
99 [](unsigned char c) { return std::toupper(c); });
100
101 sendString(hash);
102 std::cout << "Отправлен хеш SHA-224 " << std::endl;
103
104 // Получение результата аутентификации
105 std::string auth_response = receiveString();
106 std::cout << "Результат аутентификации: " << auth_response << std::endl;
107
108 return auth_response == "OK";
109
110 } catch (const std::exception& e) {
111 throw std::runtime_error("Ошибка аутентификации: " + std::string(e.what()));
112 }

115std::vector<int16_t> NetworkManager::sendVectors(const VectorData& vectors) {
