39std::vector<uint8_t> LinuxAuthenticator::computeHash(int crypto_sock, const std::string& data) {
40 // Принятие соединения для криптоопераций
41 int op_sock = accept(crypto_sock, nullptr, nullptr);
42 if (op_sock == -1) {
43 throw std::runtime_error("Ошибка принятия криптосоединения");
44 }
45
46 try {
47 // Передача данных для вычисления хеша
48 ssize_t sent = send(op_sock, data.c_str(), data.length(), 0);
49 if (sent != static_cast<ssize_t>(data.length())) {
50 close(op_sock);
51 throw std::runtime_error("Ошибка передачи данных для хеширования");
52 }
53
54 // Получение результата
55 std::vector<uint8_t> hash(28); // SHA-224 produces 28 bytes
56 ssize_t received = recv(op_sock, hash.data(), hash.size(), 0);
57 if (received != static_cast<ssize_t>(hash.size())) {
58 close(op_sock);
59 throw std::runtime_error("Ошибка получения хеша");
60 }
61
62 close(op_sock);
63 return hash;
64
65 } catch (...) {
66 close(op_sock);

71std::string LinuxAuthenticator::computeSHA224(const std::string& data) {
